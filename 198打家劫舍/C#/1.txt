public class Solution 
{
    public int Rob(int[] nums) 
    {
        int len = nums.Length;
        if(nums == null||len == 0)
            return 0;
        if(len == 1)
            return nums[0];
        int[] dp = new int[len];
        dp[0] = nums[0];
        dp[1] = Math.Max(nums[0],nums[1]);
        for(int i = 2;i < len;i++)
        {
            dp[i] = Math.Max(dp[i-1],nums[i]+dp[i-2]);daj
        }
        return dp[len-1];
    }
}
