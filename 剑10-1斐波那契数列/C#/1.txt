public class Solution 
{
    public int Fib(int n) 
    {
        int fib1 = 0;
        int fib2 = 1;
        int fibNext = 0;
        if(n <= 0)
            return fib1;
        if(n == 1)
            return fib2;
        for(int i = 2;i <= n;i++)
        {
            fibNext = (fib1 + fib2)%1000000007;
            fib1 = fib2;
            fib2 = fibNext;
        }        
        return fibNext;
    }
}
